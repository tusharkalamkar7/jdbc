
CREATE TABLE 'Employee'
(
empId int NOT NULL,
firstName varchar(30),
lastName varchar(30),
mobNo int,
deptId int,
createdDateTime date,
updatedDateTime date,
createdId int,
updatedId int,
PRIMARY KEY (empId),
FOREIGN KEY (deptId) REFERENCES Department(deptId)
)

CREATE TABLE 'Address'
(
id int NOT NULL,
empId int,
line1 varchar(30),
line2 varchar(30),
city varchar(20),
state varchar2(20),
pincode int,
createdDateTime date,
updatedDateTime date,
createdId int,
updatedId int,
PRIMARY KEY (id),
FOREIGN KEY (empId) REFERENCES Employee(empId)
)

CREATE TABLE 'Department'
(
deptId int NOT NULL,
deptName varchar(30),
createdDateTime date,
updatedDateTime date,
createdId int,
updatedId int,
PRIMARY KEY (deptId)
)
