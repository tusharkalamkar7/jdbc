
CREATE TABLE 'Employee'
(
id int NOT NULL,
firstName varchar2(30),
lastName varchar2(30),
mobNo int,
deptId
createdDateTime date,
updatedDateTime date,
createdId int,
updatedId int
)

CREATE TABLE 'Address'
(
id int NOT NULL,
empId
line1 varchar2(30),
line2 varchar2(30),
city varchar2(20),
state varchar2(20),
pincode int,
createdDateTime date,
updatedDateTime date,
createdId int,
updatedId int
)

CREATE TABLE 'Department'
(
id int NOT NULL,
deptName varchar2(30),
createdDateTime date,
updatedDateTime date,
createdId int,
updatedId int
)
